version: '3.8'

services:
  postgres:
    image: postgres:15-alpine
    container_name: zaitoon_db
    environment:
      POSTGRES_DB: zaitoon_erp
      POSTGRES_USER: zaitoon_admin
      POSTGRES_PASSWORD: ${DB_PASSWORD:-zaitoon@2024}
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./backend/migrations:/docker-entrypoint-initdb.d
    ports:
      - "5432:5432"
    networks:
      - zaitoon_network
    restart: unless-stopped

  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: zaitoon_backend
    environment:
      NODE_ENV: ${NODE_ENV:-production}
      DB_HOST: postgres
      DB_PORT: 5432
      DB_NAME: zaitoon_erp
      DB_USER: zaitoon_admin
      DB_PASSWORD: ${DB_PASSWORD:-zaitoon@2024}
      JWT_SECRET: ${JWT_SECRET:-your-secret-key-here}
      TWILIO_ACCOUNT_SID: ${TWILIO_ACCOUNT_SID}
      TWILIO_AUTH_TOKEN: ${TWILIO_AUTH_TOKEN}
      TWILIO_PHONE_NUMBER: ${TWILIO_PHONE_NUMBER}
      SENDGRID_API_KEY: ${SENDGRID_API_KEY}
      SENDGRID_FROM_EMAIL: ${SENDGRID_FROM_EMAIL}
    volumes:
      - ./backend:/app
      - /app/node_modules
    ports:
      - "5000:5000"
    depends_on:
      - postgres
    networks:
      - zaitoon_network
    restart: unless-stopped
    command: npm start

  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile
    container_name: zaitoon_frontend
    environment:
      REACT_APP_API_URL: ${REACT_APP_API_URL:-http://localhost:5000/api}
    volumes:
      - ./frontend:/app
      - /app/node_modules
    ports:
      - "3000:3000"
    depends_on:
      - backend
    networks:
      - zaitoon_network
    restart: unless-stopped
    command: npm start

  nginx:
    image: nginx:alpine
    container_name: zaitoon_nginx
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf
      - ./nginx/sites-enabled:/etc/nginx/sites-enabled
    ports:
      - "80:80"
      - "443:443"
    depends_on:
      - frontend
      - backend
    networks:
      - zaitoon_network
    restart: unless-stopped

volumes:
  postgres_data:

networks:
  zaitoon_network:
    driver: bridge
